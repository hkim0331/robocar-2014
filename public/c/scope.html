変数のスコープ

変数のスコープはその変数が定義された場所から下、{ }の内部。

関数のスコープはプログラム全体。

(若干嘘だが、今の段階ではそう覚えておいて十分。)

#include <stdio.h>

int x=100;

void func1(void)
{
	printf("func1, x=%i\n",x);
}
void func2(void)
{
	int x=10;
	printf("func2, x=%i\n",x);
}
void func3(int x)
{
	print("func3, x=%i\n", x);
}
void func4(int *x)
{
	*x = 5000;
	printf("func4, x=%i, *x=%i\n", x, *x);
}
int main(void)
{
	int x;

	x=300;
	func1();
	func2();
	func3(x);
	printf("main, x=%i\n",x);
	func4(&x);
	printf("main, x=%i\n",x);
	return 0;
}
